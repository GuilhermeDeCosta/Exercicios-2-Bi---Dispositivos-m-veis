import { StyleSheet, Text, View, FlatList, ActivityIndicator, ScrollView, TouchableOpacity } from 'react-native';
import { useState, useEffect } from 'react';
import axios from 'axios';

export default function App() {
  const [isLoadingLista, setIsLoadingLista] = useState(false);
  const [isLoadingDetalhes, setIsLoadingDetalhes] = useState(false);
  
  const [arrayDados, setArrayDados] = useState([]);

  const [nome, setNome] = useState('');
  const [preco, setPreco] = useState('');
  const [descricao, setDescricao] = useState('');
  const [categoria, setCategoria] = useState('');

  useEffect(() => {
    buscarProdutos();
  }, []);

  async function buscarProdutos() {
    setIsLoadingLista(true); // Ativa o loading só da lista
    try {
      const resultados = await axios.get('https://6a2ec814c9776ca6c0c4ef8f.mockapi.io/produtos');
      setArrayDados(resultados.data);
    } catch (error) {
      console.log(error);
      alert("Não há dados");
    }
    setIsLoadingLista(false); // Desativa o loading da lista
  }

  async function buscarProdutosId(id) {
    setIsLoadingDetalhes(true); // Ativa o loading só do quadro branco
    try {
      const resultados = await axios.get('https://6a2ec814c9776ca6c0c4ef8f.mockapi.io/produtos/' + id);

      setNome(resultados.data.nome);
      setPreco(resultados.data.preco);
      setDescricao(resultados.data.descricao);
      setCategoria(resultados.data.categoria);
      
    } catch(error) {
      console.log('Erro ao buscar produto');
    } finally {
      setIsLoadingDetalhes(false); // Desativa garantido, dando erro ou não
    }
  }

  function limpar() {
    setNome('');
    setPreco('');
    setDescricao('');
    setCategoria('');
  }

  return (
    <View style={styles.container}>

      <View style={{ backgroundColor: 'purple', paddingVertical: 20, paddingHorizontal: 10 }}>
        
        <View style={{ backgroundColor: 'white', padding: 20, marginHorizontal: 20, marginBottom: 10, borderRadius: 15 }}>
          
          {isLoadingDetalhes ? (
            <ActivityIndicator size="small" color="purple" />
          ) : nome ? (
            <View>
              <Text style={{ fontWeight: 'bold', fontSize: 16, marginBottom: 5 }}>Nome: {nome}</Text>
              <Text style={{ marginBottom: 5 }}>Preço: R$ {preco}</Text>
              <Text style={{ marginBottom: 5 }}>Categoria: {categoria}</Text>
              <Text style={{ fontStyle: 'italic', color: 'grey' }}>Descrição: {descricao}</Text>
            </View>
          ) : (
            <Text style={{ textAlign: 'center', color: 'grey' }}>Nenhum produto selecionado</Text>
          )}

        </View>

        <TouchableOpacity 
          onPress={limpar} 
          style={{ backgroundColor: 'green', borderRadius: 4, padding: 10, marginHorizontal: 20, alignItems: 'center' }}
        >
          <Text style={{ color: 'white', fontWeight: 'bold' }}>Limpar</Text>
        </TouchableOpacity>

      </View>

      <ScrollView style={{ flex: 1 }}>
        <View style={{ paddingBottom: 20 }}>
          {isLoadingLista ? (
            <ActivityIndicator size="large" style={{ marginTop: 20 }} />
          ) : (
            <FlatList 
              data={arrayDados}
              renderItem={({ item }) => {
                return (
                  <View style={styles.lista}>
                    <TouchableOpacity onPress={() => buscarProdutosId(item.id)}>
                      <Text style={{ color: 'white', fontWeight: 'bold' }}>{item.id} - {item.nome}</Text>
                    </TouchableOpacity>
                  </View>
                );
              }}
              keyExtractor={(item) => item.id}
            />
          )}
        </View>
      </ScrollView>
    </View>
  );
}

const styles = StyleSheet.create({
  container: {
    flex: 1,
    marginHorizontal: 8
  },
  lista: {
    backgroundColor: 'green',
    padding: 30,
    marginVertical: 4,
    borderRadius: 30
  }
});
