import { StyleSheet, Text, View, FlatList, ActivityIndicator, ScrollView, TouchableOpacity, TextInput, Modal } from 'react-native';
import { useState, useEffect } from 'react';
import axios from 'axios';
import EvilIcons from '@expo/vector-icons/EvilIcons';
import Entypo from '@expo/vector-icons/Entypo';

export default function App() {
  const [isLoading, setIsLoading] = useState(false);
  const [isLoadingDetalhes, setIsLoadingDetalhes] = useState(false);
  const [botaoDesabilidatado, setBotaoDesabilidatado] = useState(false);
  
  const [arrayDados, setArrayDados] = useState([]);

  // Estados de exibição de detalhes
  const [nomeProduto, setNomeProduto] = useState('');
  const [precoProduto, setPrecoProduto] = useState('');
  const [descricaoProduto, setDescricaoProduto] = useState('');
  const [categoriaProduto, setCategoriaProduto] = useState('');

  // Estados de cadastro
  const [nome, setNome] = useState('');
  const [preco, setPreco] = useState('');
  const [descricao, setDescricao] = useState(''); 
  const [categoria, setCategoria] = useState('');

  // ======== NOVOS ESTADOS PARA O MODAL DE EDIÇÃO ========
  const [modalVisivel, setModalVisivel] = useState(false);
  const [idEdicao, setIdEdicao] = useState(null);
  const [nomeEdicao, setNomeEdicao] = useState('');
  const [precoEdicao, setPrecoEdicao] = useState('');
  const [descricaoEdicao, setDescricaoEdicao] = useState('');
  const [categoriaEdicao, setCategoriaEdicao] = useState('');

  useEffect(() => {
    buscarProdutos();
  }, []);

  async function buscarProdutos() {
    setIsLoading(true);
    try {
      const resultados = await axios.get('https://6a2ec814c9776ca6c0c4ef8f.mockapi.io/produtos');
      setArrayDados(resultados.data);
    } catch (error) {
      console.log(error);
      alert("Não há dados");
    }
    setIsLoading(false);
  }
 
  async function buscarProdutosId(id) {
    setIsLoadingDetalhes(true);
    try {
      const resultados = await axios.get('https://6a2ec814c9776ca6c0c4ef8f.mockapi.io/produtos/' + id);

      setNomeProduto(resultados.data.nome);
      setPrecoProduto(resultados.data.preco);
      setDescricaoProduto(resultados.data.descricao);
      setCategoriaProduto(resultados.data.categoria);
      
    } catch(error) {
      console.log('Erro ao buscar produto');
    }
    setIsLoadingDetalhes(false);
  }

  function limpar() {
    setNomeProduto('');
    setPrecoProduto('');
    setDescricaoProduto('');
    setCategoriaProduto('');
  }

  async function cadastrarNovoProduto() {
    if (!nome.trim() || !preco.trim() || !descricao.trim() || !categoria.trim()) {
      alert("Há campos faltando serem preenchidos");
      return;
    }

    setBotaoDesabilidatado(true);

    const payload = { nome, preco, descricao, categoria };

    try {
      await axios.post('https://6a2ec814c9776ca6c0c4ef8f.mockapi.io/produtos', payload);

      setNome('');
      setPreco('');
      setDescricao('');
      setCategoria('');

      buscarProdutos();
    } catch (error) {
      console.log(error);
    }
    setBotaoDesabilidatado(false);
  }

  async function deletarItem(idItem) {
    setBotaoDesabilidatado(true);
    try {
      await axios.delete('https://6a2ec814c9776ca6c0c4ef8f.mockapi.io/produtos/' + idItem);
      buscarProdutos();
    } catch (error) {
      console.log("Erro ao deletar item");
    }
    setBotaoDesabilidatado(false);
  }

  // ======== NOVAS FUNÇÕES PARA O MODAL ========
  function abrirModalEdicao(item) {
    setIdEdicao(item.id);
    setNomeEdicao(item.nome);
    setPrecoEdicao(item.preco);
    setDescricaoEdicao(item.descricao);
    setCategoriaEdicao(item.categoria);
    setModalVisivel(true);
  }

  function fecharModal() {
    setModalVisivel(false);
    setIdEdicao(null);
  }

  async function salvarEdicao() {
    if (!nomeEdicao.trim() || !precoEdicao.trim() || !descricaoEdicao.trim() || !categoriaEdicao.trim()) {
      alert("Preencha todos os campos para editar.");
      return;
    }

    setBotaoDesabilidatado(true);
    const payload = {
      nome: nomeEdicao,
      preco: precoEdicao,
      descricao: descricaoEdicao,
      categoria: categoriaEdicao
    };

    try {
      await axios.put(`https://6a2ec814c9776ca6c0c4ef8f.mockapi.io/produtos/${idEdicao}`, payload);
      alert("Produto atualizado com sucesso!");
      fecharModal();
      buscarProdutos();
    } catch (error) {
      console.log("Erro ao editar", error);
      alert("Erro ao editar o produto.");
    }
    setBotaoDesabilidatado(false);
  }

  return (
    <View style={styles.container}>

      {/* ======== COMPONENTE MODAL DE EDIÇÃO ======== */}
      <Modal
        animationType="slide"
        transparent={true}
        visible={modalVisivel}
        onRequestClose={fecharModal}
      >
        <View style={styles.modalBackground}>
          <View style={styles.modalContainer}>
            <Text style={{ fontSize: 18, fontWeight: 'bold', marginBottom: 15 }}>Editar Produto</Text>

            <Text style={styles.textoModal}>Nome do produto</Text>
            <TextInput style={styles.inputModal} value={nomeEdicao} onChangeText={setNomeEdicao} />

            <Text style={styles.textoModal}>Preço do produto</Text>
            <TextInput style={styles.inputModal} value={precoEdicao} onChangeText={setPrecoEdicao} keyboardType="numeric" />

            <Text style={styles.textoModal}>Descrição do produto</Text>
            <TextInput style={styles.inputModal} value={descricaoEdicao} onChangeText={setDescricaoEdicao} />

            <Text style={styles.textoModal}>Categoria do produto</Text>
            <TextInput style={styles.inputModal} value={categoriaEdicao} onChangeText={setCategoriaEdicao} />

            <View style={{ flexDirection: 'row', justifyContent: 'space-between', marginTop: 10 }}>
              <TouchableOpacity onPress={fecharModal} style={[styles.btnModal, { backgroundColor: 'red' }]}>
                <Text style={{ color: 'white', fontWeight: 'bold' }}>Cancelar</Text>
              </TouchableOpacity>

              <TouchableOpacity onPress={salvarEdicao} disabled={botaoDesabilidatado} style={[styles.btnModal, { backgroundColor: 'green' }]}>
                <Text style={{ color: 'white', fontWeight: 'bold' }}>Salvar</Text>
              </TouchableOpacity>
            </View>
          </View>
        </View>
      </Modal>


      <View style={{ backgroundColor: 'purple', paddingVertical: 20, paddingHorizontal: 10 }}>
        <View style={{ backgroundColor: 'white', padding: 20, marginHorizontal: 20, marginBottom: 10, borderRadius: 15 }}>
          
          {isLoadingDetalhes ? (
            <ActivityIndicator size="small" color="purple" />
          ) : nomeProduto ? (
            <View>
              <Text style={{ fontWeight: 'bold', fontSize: 16, marginBottom: 5 }}>Nome: {nomeProduto}</Text>
              <Text style={{ marginBottom: 5 }}>Preço: R$ {precoProduto}</Text>
              <Text style={{ marginBottom: 5 }}>Categoria: {categoriaProduto}</Text>
              <Text style={{ fontStyle: 'italic', color: 'grey' }}>Descrição: {descricaoProduto}</Text>
            </View>
          ) : (
            <Text style={{ textAlign: 'center', color: 'grey' }}>Nenhum produto selecionado</Text>
          )}

        </View>

        <TouchableOpacity 
          onPress={limpar} 
          style={{ backgroundColor: 'green', borderRadius: 4, padding: 10, marginHorizontal: 20, alignItems: 'center' }}
        >
          <Text style={{ color: 'white', fontWeight: 'bold' }}>Limpar Seleção</Text>
        </TouchableOpacity>

        <View style={{marginHorizontal:10, marginVertical: 20}}>
          <Text style={styles.textoInput}>Nome do produto</Text>
          <TextInput style={styles.input} value={nome} onChangeText={setNome} />

          <Text style={styles.textoInput}>Preço do produto</Text>
          <TextInput style={styles.input} value={preco} onChangeText={setPreco} keyboardType="numeric" />

          <Text style={styles.textoInput}>Descrição do produto</Text>
          <TextInput style={styles.input} value={descricao} onChangeText={setDescricao} />

          <Text style={styles.textoInput}>Categoria do produto</Text>
          <TextInput style={styles.input} value={categoria} onChangeText={setCategoria} />

          <TouchableOpacity 
            onPress={cadastrarNovoProduto}
            disabled={botaoDesabilidatado}
            style={{padding: 15, backgroundColor:'green', borderRadius:10, alignItems: 'center'}}>
            <Text style={{color:'white', fontWeight: 'bold'}}>Cadastrar produto</Text>
          </TouchableOpacity>
        </View>
      </View>

      <ScrollView style={{ flex: 1 }}>
        <View style={{ paddingBottom: 20 }}>
          {isLoading ? (
            <ActivityIndicator size="large" style={{ marginTop: 20 }} />
          ) : (
            <FlatList 
              data={arrayDados}
              renderItem={({ item }) => {
                return (
                  <View style={styles.lista}>
                    <TouchableOpacity onPress={() => buscarProdutosId(item.id)} style={{ flex: 1 }}>
                      <Text style={{ color: 'white', fontWeight: 'bold' }}>{item.id} - {item.nome}</Text>
                    </TouchableOpacity>

                    <View style={{ flexDirection: 'row', alignItems: 'center' }}>
                      {/* Botão de abrir modal de edição */}
                      <TouchableOpacity onPress={() => abrirModalEdicao(item)} style={{ marginRight: 15 }}>
                        <Entypo name="pencil" size={24} color="white" />
                      </TouchableOpacity>

                      <TouchableOpacity onPress={() => deletarItem(item.id)}>
                        <EvilIcons name="trash" size={30} color="white" />
                      </TouchableOpacity>
                    </View>
                  </View>
                );
              }}
              keyExtractor={(item) => item.id}
            />
          )}
        </View>
      </ScrollView>
    </View>
  );
}

const styles = StyleSheet.create({
  container: { flex: 1, marginHorizontal: 8 },
  lista: { backgroundColor: 'green', padding: 20, marginVertical: 4, borderRadius: 15, flexDirection: 'row', justifyContent: 'space-between', alignItems: 'center' },
  input: { backgroundColor: '#f2f2f2', padding: 10, marginBottom: 10, borderRadius: 10 },
  textoInput: { color: 'white', marginVertical:0.5, fontWeight: 'bold' },
  
  // Estilos do Modal
  modalBackground: { flex: 1, justifyContent: 'center', alignItems: 'center', backgroundColor: 'rgba(0,0,0,0.6)' },
  modalContainer: { backgroundColor: 'white', padding: 20, borderRadius: 15, width: '90%', elevation: 5 },
  inputModal: { backgroundColor: '#f2f2f2', padding: 10, marginBottom: 15, borderRadius: 10, borderWidth: 1, borderColor: '#ddd' },
  textoModal: { fontWeight: 'bold', marginBottom: 5 },
  btnModal: { flex: 1, padding: 15, borderRadius: 10, alignItems: 'center', marginHorizontal: 5 }
});
